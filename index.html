} // ── CheckMacValue 產生（ECPay 規格）────────────────────────────── function makeMac(params, HashKey, HashIV) { const sorted = Object.keys(params).sort((a, b) => a.toLowerCase().localeCompare(b.toLowerCase())); return crypto.createHash("sha256").update(enc).digest("hex").toUpperCase(); } function verifyMac(params, HashKey, HashIV) { const { CheckMacValue, ...rest } = params; return makeMac(rest, HashKey, HashIV) === CheckMacValue; p.name : id; }).join('、'); throw new HttpsError("failed-precondition", `此優惠碼僅適用於：${applyNames}`); } if (c.perUserOnce !== false) { const used = await db.collection("coupons").doc(code).collection("usedBy").doc(uid).get(); if (used.exists) throw new HttpsError("failed-precondition", "您已使用過此優惠碼"); } if (c.type === "percent") finalPrice = Math.round(plan.price * (1 - c.value / 100)); else finalPrice = Math.max(0, plan.price - c.value); if (finalPrice < 1) finalPrice = 1; // ECPay 最低 1 元 } // 計算到期日（訂閱期 + 加送月） const months = (PERIOD_MONTHS[plan.period] || 1) + (plan.bonusMonths || 0); const expiresAt = Date.now() + months * 30 * 24 * 60 * 60 * 1000; // 產生訂單號（唯一） const tradeNo = "WS" + Date.now().toString(36).toUpperCase() + uid.slice(0, 4).toUpperCase(); // 寫入 pending subscription（webhook 收到付款成功後才 activate） await db.collection("users").doc(uid).collection("subscriptions").doc(tradeNo).set({ tradeNo, planId, planName: plan.name, originalPrice: plan.price, finalPrice, couponCode: couponCode || null, period: plan.period, bonusMonths: plan.bonusMonths || 0, status: "pending", expiresAt, createdAt: Date.now(), }); // 組 ECPay 定期定額參數 const ecpay = getEcpay(); const PeriodMap = { month: "M", quarter: "Q", "6month": "B", year: "Y" }; const params = { MerchantID : ecpay.MerchantID, MerchantTradeNo : tradeNo, MerchantTradeDate : new Date().toLocaleString("zh-TW", { timeZone: "Asia/Taipei", hour12: false }) .replace(/\//g, "/").slice(0, 19), TotalAmount : String(finalPrice), TradeDesc : encodeURIComponent(`${plan.name} - 婚禮 SaaS`), ItemName : `${plan.name}${ couponCode ?
